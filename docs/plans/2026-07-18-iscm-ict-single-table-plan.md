# ISCM_ICT 3테이블 → 1테이블 통합 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** ISCM_ICT 로그를 3개 테이블(LOG_ISCM_ICT / _HDATA / _COMP) 분산 저장에서 단일 테이블 `LOG_ISCM_ICT` (Component 행수만큼 N행) 로 통합한다.

**Architecture:** 헤더블록(`[KEY]:VALUE`)에서 5종 그룹(BARCODE, TOTAL_RESULT, RESULT_CURRENT, HEX_FILE_1~16, CRC_1~16)만 추출해 Component_Data 각 행 앞에 병합한다. Header_Data 섹션은 폐기. 적재는 기존 `BARCODE_REPLACE_TABLES` 경로(BARCODE 단위 DELETE → bulk INSERT)를 그대로 쓴다(PRESSFIT/LOG_ICT 와 동일 패턴).

**Tech Stack:** Vector VRL (aggregator TOML), Fastify + TypeScript backend, Oracle DDL/트리거.

## Global Constraints

- VRL 규칙: `split!`/`replace!` 의 `!` 유지 / `to_string!` + `??` 조합 금지(fallible `to_string` + `??` 사용) / `while`·`loop`·`break` 금지(반복은 `for_each`) / 동적 인덱스 접근은 정수 리터럴만(`get!(arr,[0])`) — **문자열 리터럴 키 `get(map, ["KEY"])` 는 허용**, 변수 키 `get(map, [var])` 는 피한다. `set!(map, [var], v)` 는 기존 코드에서 동작 확인됨.
- `LOG_ISCM_ICT` 는 자기 테이블을 UPDATE 하는 트리거가 없어야 한다(ORA-04091 mutating 회피 → bulk INSERT 가능). IS_LAST 는 트리거가 상수 `'Y'` 로만 세팅.
- aggregator TOML 변경은 git push 로 자동 reload 안 됨 → 반드시 `POST /api/monitor/vector/reload`.
- 운영 DB 는 SVEHICLEPDB 사이트. registry 는 API(`POST /api/monitor/registry`)로만 운영 반영되나, ISCM_ICT 는 registry 미등록(VRL 동적 컬럼)이라 이번 작업에 registry 변경 없음.
- 로컬에서 운영 서버(20.10.30.112) 명령 실행 금지. 서버 불필요 재시작 금지.

```toml
} else if .equipment_type == "ISCM_ICT" {
  # ── ISCM 모델 ICT 검사설비 (eTR5001SII) — 단일 테이블 ──
  # 로그 1파일 = 헤더블록([KEY]:VALUE) + // Header_Data + // Component_Data.
  #   Header_Data 는 폐기한다. 헤더블록에서 5종 그룹만 추출해 Component 각 행 앞에
  #   병합 → 단일 테이블 LOG_ISCM_ICT (Component 행수만큼 N행).
  #
  # 추출 컬럼:
  #   BARCODE, TOTAL_RESULT([RESULT]), RESULT_CURRENT([RESULT CURRENT]),
  #   HEX_FILE_1..16([HEX FILE n]), CRC_1..16([HEX CRC32 n] → CRC_n).
  # 중복: backend BARCODE_REPLACE_TABLES 가 BARCODE 단위 DELETE 후 bulk INSERT.
  # 날짜: 헤더에 신뢰 가능한 날짜 없음 → 트리거가 SYSDATE 로 ACTUAL_DATE 채움.

  lines = split!(.message, "\n")

  barcode = ""
  section = "HEADER"
  hdr = {}
  comp_lines = []

  for_each(array!(lines)) -> |_idx, raw| {
    line = to_string!(raw)
    t = strip_whitespace(line)

    if starts_with(t, "// Header_Data") {
      section = "HDATA"
    } else if starts_with(t, "// Component_Data") {
      section = "COMP"
    } else if starts_with(t, "//") || t == "" || starts_with(t, "*") {
      null  # 주석/빈줄/구분선 무시 (HDATA 섹션 본문도 여기서 흘려보냄)
    } else if section == "HEADER" {
      # BARCODE : [ 6711001861B ]
      bm = parse_regex(t, r'^BARCODE\s*:\s*\[\s*(?P<b>.*?)\s*\]\s*$') ?? {}
      if is_string(get(bm, ["b"]) ?? null) {
        barcode = to_string!(get!(bm, ["b"]))
      } else {
        # [KEY]:VALUE  /  [KEY   ] : VALUE
        km = parse_regex(t, r'^\[(?P<k>[^\]]+)\]\s*:(?P<v>.*)$') ?? {}
        if is_string(get(km, ["k"]) ?? null) {
          k = strip_whitespace(to_string!(get!(km, ["k"])))
          k = replace(k, "#", "")
          k = upcase(replace(k, r'\s+', "_"))
          v = strip_whitespace(to_string!(get!(km, ["v"])))
          # 후행 슬래시 정리 (예: "TEST_ISCM / ")
          v = strip_whitespace(replace(v, r'\s*/\s*$', ""))
          hdr = set!(hdr, [k], v)
        }
      }
    } else if section == "COMP" {
      comp_lines = push(comp_lines, line)
    }
  }

  # 바코드가 없으면 로그 본문이 아님 → 빈 row INSERT 방지 위해 버림
  if barcode == "" {
    abort
  }

  # ── 헤더블록 5종 그룹 추출 (문자열 리터럴 키만 사용) ──
  head = {
    "BARCODE": barcode,
    "TOTAL_RESULT": to_string(get(hdr, ["RESULT"]) ?? "") ?? "",
    "RESULT_CURRENT": to_string(get(hdr, ["RESULT_CURRENT"]) ?? "") ?? "",
    "HEX_FILE_1": to_string(get(hdr, ["HEX_FILE_1"]) ?? "") ?? "",
    "HEX_FILE_2": to_string(get(hdr, ["HEX_FILE_2"]) ?? "") ?? "",
    "HEX_FILE_3": to_string(get(hdr, ["HEX_FILE_3"]) ?? "") ?? "",
    "HEX_FILE_4": to_string(get(hdr, ["HEX_FILE_4"]) ?? "") ?? "",
    "HEX_FILE_5": to_string(get(hdr, ["HEX_FILE_5"]) ?? "") ?? "",
    "HEX_FILE_6": to_string(get(hdr, ["HEX_FILE_6"]) ?? "") ?? "",
    "HEX_FILE_7": to_string(get(hdr, ["HEX_FILE_7"]) ?? "") ?? "",
    "HEX_FILE_8": to_string(get(hdr, ["HEX_FILE_8"]) ?? "") ?? "",
    "HEX_FILE_9": to_string(get(hdr, ["HEX_FILE_9"]) ?? "") ?? "",
    "HEX_FILE_10": to_string(get(hdr, ["HEX_FILE_10"]) ?? "") ?? "",
    "HEX_FILE_11": to_string(get(hdr, ["HEX_FILE_11"]) ?? "") ?? "",
    "HEX_FILE_12": to_string(get(hdr, ["HEX_FILE_12"]) ?? "") ?? "",
    "HEX_FILE_13": to_string(get(hdr, ["HEX_FILE_13"]) ?? "") ?? "",
    "HEX_FILE_14": to_string(get(hdr, ["HEX_FILE_14"]) ?? "") ?? "",
    "HEX_FILE_15": to_string(get(hdr, ["HEX_FILE_15"]) ?? "") ?? "",
    "HEX_FILE_16": to_string(get(hdr, ["HEX_FILE_16"]) ?? "") ?? "",
    "CRC_1": to_string(get(hdr, ["HEX_CRC32_1"]) ?? "") ?? "",
    "CRC_2": to_string(get(hdr, ["HEX_CRC32_2"]) ?? "") ?? "",
    "CRC_3": to_string(get(hdr, ["HEX_CRC32_3"]) ?? "") ?? "",
    "CRC_4": to_string(get(hdr, ["HEX_CRC32_4"]) ?? "") ?? "",
    "CRC_5": to_string(get(hdr, ["HEX_CRC32_5"]) ?? "") ?? "",
    "CRC_6": to_string(get(hdr, ["HEX_CRC32_6"]) ?? "") ?? "",
    "CRC_7": to_string(get(hdr, ["HEX_CRC32_7"]) ?? "") ?? "",
    "CRC_8": to_string(get(hdr, ["HEX_CRC32_8"]) ?? "") ?? "",
    "CRC_9": to_string(get(hdr, ["HEX_CRC32_9"]) ?? "") ?? "",
    "CRC_10": to_string(get(hdr, ["HEX_CRC32_10"]) ?? "") ?? "",
    "CRC_11": to_string(get(hdr, ["HEX_CRC32_11"]) ?? "") ?? "",
    "CRC_12": to_string(get(hdr, ["HEX_CRC32_12"]) ?? "") ?? "",
    "CRC_13": to_string(get(hdr, ["HEX_CRC32_13"]) ?? "") ?? "",
    "CRC_14": to_string(get(hdr, ["HEX_CRC32_14"]) ?? "") ?? "",
    "CRC_15": to_string(get(hdr, ["HEX_CRC32_15"]) ?? "") ?? "",
    "CRC_16": to_string(get(hdr, ["HEX_CRC32_16"]) ?? "") ?? "",
  }

  # ── Component_Data (12필드 CSV N행) + 헤더 병합 ──
  rows = []
  seq = 0
  for_each(array!(comp_lines)) -> |_i, row| {
    cols = split(to_string!(row), ",")
    seq = seq + 1
    item = merge(head, {
      "SEQ_NO": to_string(seq),
      "PART_NAME": strip_whitespace(to_string!(get(cols, [0]) ?? "")),
      "ACT_V": strip_whitespace(to_string!(get(cols, [1]) ?? "")),
      "STD_V": strip_whitespace(to_string!(get(cols, [2]) ?? "")),
      "MSR_V": strip_whitespace(to_string!(get(cols, [3]) ?? "")),
      "HLIM": strip_whitespace(to_string!(get(cols, [4]) ?? "")),
      "LLIM": strip_whitespace(to_string!(get(cols, [5]) ?? "")),
      "HPIN": strip_whitespace(to_string!(get(cols, [6]) ?? "")),
      "LPIN": strip_whitespace(to_string!(get(cols, [7]) ?? "")),
      "PART_TYPE": strip_whitespace(to_string!(get(cols, [8]) ?? "")),
      "RESULT": strip_whitespace(to_string!(get(cols, [9]) ?? "")),
      "DEV": strip_whitespace(to_string!(get(cols, [10]) ?? "")),
      "SKIP_FLAG": strip_whitespace(to_string!(get(cols, [11]) ?? "")),
    })
    rows = push(rows, item)
  }

  # ── 단일 테이블 적재 (BARCODE_REPLACE_TABLES 경로) ──
  .target_table = "LOG_ISCM_ICT"
  .data = { "ROWS": rows }
} else if .equipment_type == "FCT" {
```
